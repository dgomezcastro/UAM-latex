\section{Entrega de ejercicios}

\begin{frame}{Ejercicios a entregar}

\begin{enumerate}
    \item Instala \LaTeX~ en tu equipo personal
    \item Crear el fichero \texttt{semana01.tex} que al compilar dé lugar a semana01.pdf.

\end{enumerate}
\end{frame}
