\newcommand{\latexinputandcompiledifferentdimensions}[3]
{
  \begin{columns}
    \column{#1\textwidth}
    \begin{block}{Código}
        \latexfile{#3}
    \end{block}
    \column{#2\textwidth}
    \centering
    \fbox{
    \begin{minipage}{\textwidth}
        \input{#3}
    \end{minipage}
    }
  \end{columns}
}

\newcommand{\latexinputandpdfdifferentdimensions}[3]
{
  \begin{columns}
    \column{#1\textwidth}
    \begin{block}{Código}
        \latexfile{#3.tex}
    \end{block}
    \column{#2\textwidth}
    \centering
    \fbox{
    \begin{minipage}{\textwidth}
        \includegraphics[width=\textwidth]{#3.pdf}
    \end{minipage}
    }
  \end{columns}
}

\usepackage{xpatch}

\xpatchcmd\inputminted
  {#3}
  {\import@path #3}
  {}{\fail}

% fix for second kind
\xpretocmd\minted@pygmentize
  {\restore@IfFileExists} % direct \let\IfFileExists\im@@IfFileExists works too
  {}{\fail}
\def\restore@IfFileExists{%
  \ifx\IfFileExists\@iffileonpath
    \let\IfFileExists\im@@IfFileExists
  \fi
}
