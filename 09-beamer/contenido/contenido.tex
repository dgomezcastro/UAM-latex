\begin{frame}[fragile]
    \frametitle{Las diapositivas}
    Las diapositivas se introducen en un entorno \texttt{frame}. \only<1->{Algunas cosas que se pueden añadir (editando la cabecera acordemente)}
    \only<1>{
        \begin{block}{Código}
        \latexfile[firstline=13]{codigos/beamer1.tex}
        \end{block}}
    \only<2->{
    \begin{columns}
        \column{0.3\textwidth}
            \begin{enumerate}
                \item<2-> Título
                \item<3-> Imágenes
                \item<4-> Teoremas
            \end{enumerate}
        \column{0.7\textwidth}
        \footnotesize
        \only<2>{
            \begin{block}{Código}
                 \latexfile[firstline=21,lastline=23]{codigos/beamer2.tex}
            \end{block}}
        \only<3>{
            \begin{block}{Código}
                \latexfile[firstline=25,lastline=32]{codigos/beamer2.tex}
            \end{block}}
        \only<4>{
            \begin{block}{Código}
                \latexfile[firstline=34,lastline=41]{codigos/beamer2.tex}
            \end{block}}
    \end{columns}}
\end{frame}
