\section{Entrega de ejercicio}

\begin{frame}{Ejercicio a entregar}
    Utilizar \texttt{tikz} para crear una figura o diagrama. Debe contener, al menos:
    \begin{enumerate}
        \item Una flecha
        \item Un segmento de algún color distinto del negro
        \item Un nodo con texto o una fórmula
    \end{enumerate}
\end{frame}
