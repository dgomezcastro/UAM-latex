\section{Entrega de ejercicios}

\begin{frame}{Ejercicio a entregar}
    Hacer un juego de diapositivas que contenga, al menos:
    \begin{enumerate}
        \item Portada con título y nombre
        \item Tabla de contenidos
        \item Una sección
        \item Un teorema o un bloque
        \item Una figura
        \item Una referencia bibliográfica con bibtex o biblatex.
    \end{enumerate}
\end{frame}
