\section{Entrega de ejercicios}

\begin{frame}{Ejercicio a entregar}
    Entregar un documento \texttt{semana10.tex}:
    \begin{enumerate} 
        \item Cada uno de los siguientes ejercicios en un archivo .tex distintos, con un \latexinline{\input} en \texttt{semana10.tex}.

        \item Un nuevo comando \latexinline{\fraccion} que reciba dos números y tal que
        
        \latexinline{\fraccion{8}{5}} 
        devuelva
        \boxed{\begin{tabular}{rr}  8 \\ $\times$ 5 \\ \hline 40 \end{tabular}}
        
        \item 
        Crear un entorno personalizado de ejercicio, con un contador que se numere relativo a la sección en la que se encuentra 
        (no utilizar librerías).

        
        \item 
        Un entorno personalizado de teorema, que se numere con su propio contador y sólo utilice números impares.

        (Puntos adicionales si los teoremas aparecen enmarcados)
    \end{enumerate}
\end{frame}
